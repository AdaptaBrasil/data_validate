data_validate.helpers.base.data_args

This module provides classes for argument parsing, validation, and configuration used in the Adapta Parser project.

Classes: DataModelABC: Abstract base class for argument parsing and validation. DataFile: Handles file-related arguments and operations. DataAction: Handles action-related arguments and operations. DataReport: Handles report-related arguments and operations. DataArgs: Main class for parsing and managing all program arguments.

 20class DataModelABC(ABC):
 21    """
 22    Abstract base class for argument parsing and validation.
 23
 24    Methods:

 26    """

 28    def __init__(self):
 29        """
 30        Initializes the DataModelABC class.

 34    @abstractmethod
 35    def _validate_arguments(self):
 36        """
 37        Validates the parsed arguments.

 40            NotImplementedError: If the method is not implemented in a subclass.

 45class DataFile(DataModelABC):
 46    """
 47    Handles file-related arguments and operations.
 48
 49    Attributes:
 50        input_folder (str): Path to the input folder.
 51        output_folder (str): Path to the output folder.
 52        locale (str): Locale setting.
 53
 54    Methods:
 55        _validate_arguments(): Validates the file-related arguments.
 56        run(): Parses and validates the arguments.
 57    """
 58
 59    def __init__(self, input_folder=None, output_folder=None, locale=None):
 60        """
 61        Initializes the DataFile class with default attributes.
 62
 63        Args:
 64            input_folder (str, optional): Path to the input folder.
 65            output_folder (str, optional): Path to the output folder.
 66            locale (str, optional): Locale setting.
 67        """
 68        super().__init__()
 69        self.input_folder = input_folder
 70        self.output_folder = output_folder
 71        self.locale = locale
 72
 73        # Run the argument parser
 74        self.run()
 75
 76    def _validate_arguments(self):
 77        """
 78        Validates the file-related arguments.
 79
 80        Raises:
 81            ValueError: If the input folder does not exist or the output folder name is invalid.
 82        """
 83        if not os.path.isdir(self.input_folder):
 84            raise ValueError(f"Input folder does not exist: {self.input_folder}")
 85
 86        if os.path.splitext(os.path.basename(self.output_folder))[
 87            1
 88        ] != "" or "." in os.path.basename(self.output_folder):
 89            raise ValueError(f"Output folder name is invalid: {self.output_folder}")
 90
 91    def run(self):
 92        """
 93        Parses and validates the file-related arguments.
 94        """
 95        self._validate_arguments()

 98class DataAction(DataModelABC):
 99    """
100    Handles action-related arguments and operations.
101
102    Attributes:
103        no_spellchecker (bool): Indicates whether the spell checker should be disabled.
104        no_warning_titles_length (bool): Indicates whether warnings for title length should be disabled.
105        no_time (bool): Indicates whether execution time and date information should be hidden.
106        no_version (bool): Indicates whether the script version should be hidden in the final report.
107        debug (bool): Indicates whether the program should run in debug mode.
108
109    Methods:
110        __init__(no_spellchecker=None, no_warning_titles_length=None, no_time=None, no_version=None, debug=None):
111            Initializes the DataAction class with the provided arguments.
112        _validate_arguments():
113            Validates the action-related arguments.
114        run():
115            Parses and validates the command-line arguments.
116    """
117
118    def __init__(
119        self,
120        no_spellchecker=None,
121        no_warning_titles_length=None,
122        no_time=None,
123        no_version=None,
124        debug=None,
125    ):
126        """
127        Initializes the DataAction class with default attributes.
128
129        Args:
130            no_spellchecker (bool, optional): Disables the spell checker. Defaults to None.
131            no_warning_titles_length (bool, optional): Disables warnings for title length. Defaults to None.
132            no_time (bool, optional): Hides execution time and date information. Defaults to None.
133            no_version (bool, optional): Hides the script version in the final report. Defaults to None.
134            debug (bool, optional): Runs the program in debug mode. Defaults to None.
135        """
136        super().__init__()
137        self.no_spellchecker = no_spellchecker
138        self.no_warning_titles_length = no_warning_titles_length
139        self.no_time = no_time
140        self.no_version = no_version
141        self.debug = debug
142
143        # Run the argument parser
144        self.run()
145
146    def _validate_arguments(self):
147        """
148        Validates the parsed arguments.
149
150        Raises:
151            ValueError: If any of the arguments are not of the expected type.
152        """
153        if not isinstance(self.no_spellchecker, bool):
154            raise ValueError("no_spellchecker must be a boolean value.")
155        if not isinstance(self.no_warning_titles_length, bool):
156            raise ValueError("no_warning_titles_length must be a boolean value.")
157        if not isinstance(self.no_time, bool):
158            raise ValueError("no_time must be a boolean value.")
159        if not isinstance(self.no_version, bool):
160            raise ValueError("no_version must be a boolean value.")
161        if not isinstance(self.debug, bool):
162            raise ValueError("debug must be a boolean value.")
163
164    def run(self):
165        """
166        Parses and validates the command-line arguments.
167        """
168        self._validate_arguments()

225class DataArgs:
226    """
227    A class to handle argument parsing, configuration, and validation.
228
229    Attributes:
230        data_file (DataFile): Instance for handling file-related arguments.
231        data_action (DataAction): Instance for handling action-related arguments.
232        data_report (DataReport): Instance for handling report-related arguments.
233        allow_abbrev (bool): Indicates if argument abbreviations are allowed.
234
235    Methods:
236        _create_parser(): Creates and configures the argument parser.
237        get_dict_args(): Returns the parsed arguments as a dictionary.
238        __str__(): Returns a string representation of the parsed arguments.
239        run(): Parses and validates the command-line arguments.
240    """
241
242    def __init__(self, allow_abbrev=True):
243        """
244        Initializes the DataArgs class with default attributes.
245
246        Args:
247            allow_abbrev (bool, optional): Allows argument abbreviations. Defaults to True.
248        """
249        self.data_file = None
250        self.data_action = None
251        self.data_report = None
252        self.allow_abbrev = allow_abbrev
253
254        # Run the argument parser
255        self.run()
256
257    def _create_parser(self):
258        """
259        Creates an argument parser with the required arguments.
260
261        Returns:
262            argparse.ArgumentParser: The configured argument parser.
263        """
264        parser = argparse.ArgumentParser(
265            description="Adapta Parser - Processes the program arguments.",
266            allow_abbrev=self.allow_abbrev,
267        )
268
269        # Arguments for DataFile
270        parser.add_argument(
271            "--input_folder", type=str, required=True, help="Path to the input folder."
272        )
273        parser.add_argument(
274            "--output_folder",
275            default="output_data/",
276            type=str,
277            help="Path to the output folder.",
278        )
279        parser.add_argument(
280            "--locale",
281            "-l",
282            type=str,
283            choices=["pt_BR", "en_US"],
284            default="pt_BR",
285            help="Sets the locale (pt_BR or en_US).",
286        )
287
288        # Arguments for DataAction
289        parser.add_argument(
290            "--no-spellchecker", action="store_true", help="Disables the spell checker."
291        )
292        parser.add_argument(
293            "--no-warning-titles-length",
294            action="store_true",
295            help="Disables warnings for title length.",
296        )
297        parser.add_argument(
298            "--no-time",
299            action="store_true",
300            help="Hides execution time and date information.",
301        )
302        parser.add_argument(
303            "--no-version",
304            action="store_true",
305            help="Hides the script version in the final report.",
306        )
307        parser.add_argument(
308            "--debug", action="store_true", help="Runs the program in debug mode."
309        )
310
311        # Arguments for DataReport
312        parser.add_argument(
313            "--sector", type=str, default=None, help="Name of the strategic sector."
314        )
315        parser.add_argument(
316            "--protocol", type=str, default=None, help="Name of the protocol."
317        )
318        parser.add_argument("--user", type=str, default=None, help="Name of the user.")
319        parser.add_argument(
320            "--file", type=str, default=None, help="Name of the file to be analyzed."
321        )
322
323        return parser
324
325    def get_dict_args(self):
326        """
327        Returns the parsed arguments as a dictionary.
328
329        Returns:
330            dict: Dictionary of parsed arguments.
331        """
332        return {
333            "input_folder": self.data_file.input_folder,
334            "output_folder": self.data_file.output_folder,
335            "locale": self.data_file.locale,
336            "no_spellchecker": self.data_action.no_spellchecker,
337            "no_warning_titles_length": self.data_action.no_warning_titles_length,
338            "no_time": self.data_action.no_time,
339            "no_version": self.data_action.no_version,
340            "debug": self.data_action.debug,
341            "sector": self.data_report.sector,
342            "protocol": self.data_report.protocol,
343            "user": self.data_report.user,
344            "file": self.data_report.file,
345        }
346
347    def __str__(self):
348        """
349        Returns a string representation of the parsed arguments.
350
351        Returns:
352            str: String representation of the parsed arguments.
353        """
354        return (
355            f"DataArgs(input_folder={self.data_file.input_folder}, "
356            f"output_folder={self.data_file.output_folder}, locale={self.data_file.locale}, "
357            f"no_spellchecker={self.data_action.no_spellchecker}, "
358            f"no_warning_titles_length={self.data_action.no_warning_titles_length}, "
359            f"no_time={self.data_action.no_time}, no_version={self.data_action.no_version}, "
360            f"debug={self.data_action.debug}, sector={self.data_report.sector}, "
361            f"protocol={self.data_report.protocol}, user={self.data_report.user}, "
362            f"file={self.data_report.file})"
363        )
364
365    def run(self):
366        """
367        Parses and validates the command-line arguments.
368        """
369        # Create argument parser
370        parser = self._create_parser()
371
372        # Parse arguments
373        args = parser.parse_args()
374
375        # Set attributes: DataFile, DataAction, DataReport
376        self.data_file = DataFile(args.input_folder, args.output_folder, args.locale)
377        self.data_action = DataAction(
378            args.no_spellchecker,
379            args.no_warning_titles_length,
380            args.no_time,
381            args.no_version,
382            args.debug,
383        )
384        self.data_report = DataReport(args.sector, args.protocol, args.user, args.file)
