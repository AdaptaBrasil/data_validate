 24    def detect_encoding(self, file_path: str, num_bytes: int = 1024) -> Tuple[bool, str]:
 25        """
 26        Detects the encoding of a file by reading a specified number of bytes.
 27
 28        Args:
 29            file_path (str): The path to the file.
 30            num_bytes (int): The number of bytes to read for encoding detection. Default is 1024.
 31
 32        Returns:
 33            Tuple[bool, str]: A tuple containing a boolean indicating success or failure
 34                              and the detected encoding or a translated error message.
 35        """
 36        try:
 37            if not file_path:
 38                return False, self.lm.text("fs_utils_error_file_path_empty")
 39            if not os.path.exists(file_path):
 40                return False, self.lm.text(
 41                    "fs_utils_error_file_not_found",
 42                    filename=os.path.basename(file_path),
 43                )
 44            if not os.path.isfile(file_path):
 45                return False, self.lm.text("fs_utils_error_path_not_file", path=file_path)
 46
 47            with open(file_path, "rb") as f:
 48                raw_data = f.read(num_bytes)
 49                result = chardet.detect(raw_data)
 50                encoding = result.get("encoding")
 51                if not encoding:
 52                    return False, self.lm.text("fs_utils_error_encoding_failed")
 53                return True, encoding
 54        except OSError as e:
 55            return False, self.lm.text("fs_utils_error_encoding_os", error=str(e))
 56        except Exception as e:
 57            return False, self.lm.text("fs_utils_error_unexpected", error=str(e))

 65    def configure_logger(
 66        self,
 67        logger_name: str,
 68        level: Optional[int] = None,
 69        log_file: Optional[str] = None,
 70    ) -> logging.Logger:
 71        """
 72        Configures a logger with the specified name, level, and optional log file.
 73
 74        Args:
 75            logger_name (str): The name of the logger.
 76            level (Optional[int]): The logging level. If not provided, the default level is used.
 77            log_file (Optional[str]): The path to the log file. If provided, logs will also be written to this file.
 78
 79        Returns:
 80            logging.Logger: The configured logger instance.
 81        """
 82        logger = logging.getLogger(logger_name)
 83        logger.setLevel(level or self.default_level)
 84
 85        # Console handler
 86        console_handler = logging.StreamHandler()
 87
 88        # Formatter for log messages
 89        console_handler.setFormatter(CustomFormatter())
 90        logger.addHandler(console_handler)
 91
 92        # File handler (if log_file is provided)
 93        if log_file:
 94            file_handler = logging.FileHandler(log_file)
 95            file_handler.setFormatter(CustomFormatter())
 96            logger.addHandler(file_handler)
 97
 98        return logger
