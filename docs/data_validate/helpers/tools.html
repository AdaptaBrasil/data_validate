 86    def __init__(self, input_dir: str):
 87        self.input_dir = Path(input_dir)
 88        self.scanner = FileScanner(self.input_dir)
 89        self.config = Config()

 91    @property
 92    def load_all(self):
 93        errors = []
 94        files_map, qml_files, missing_files = self.scanner.scan()
 95
 96        data = {}
 97        for name, path in files_map.items():
 98            _, header_type, _ = self.config.file_specs[name]
 99            if header_type == "single":
100                strat = SingleHeaderStrategy()
101            elif header_type == "double":
102                strat = DoubleHeaderStrategy()
103            else:
104                # qml não vai passar por aqui
105                continue
106            reader = ReaderFactory.get_reader(path, strat)
107
108            # Configure DataModel
109            df_local = None
110            try:
111                df_local = reader.read()
112            except FileNotFoundError as e:
113                errors.append(f"{path.name}: Arquivo não encontrado no diretório. Detalhes: {e} ({type(e)})")
114            except UnicodeDecodeError as e:
115                errors.append(f"{path.name}: Erro de codificação do arquivo. Verifique se está em UTF-8. Detalhes: {e} ({type(e)})")
116            except pd.errors.ParserError as e:
117                errors.append(
118                    f"{path.name}: Erro na estrutura da planilha. Verifique se há células mescladas ou formato inválido. Detalhes: {e} ({type(e)})"
119                )
120            except ValueError as e:
121                errors.append(f"{path.name}: Erro nos valores da planilha. Verifique se os tipos de dados estão corretos. Detalhes: {e} ({type(e)})")
122            except IOError as e:
123                errors.append(
124                    f"{path.name}: Erro de entrada/saída ao ler o arquivo. Verifique se ele não está aberto em outro programa. Detalhes: {e} ({type(e)})"
125                )
126            except Exception as e:
127                errors.append(f"{path.name}: Erro inesperado ao processar o arquivo. Detalhes: {e} ({type(e)})")
128
129            data_model = DataLoaderModel(
130                input_folder=str(self.input_dir),
131                path=path,
132                df_data=df_local if df_local is not None else pd.DataFrame(),
133                read_success=True if df_local is not None else False,
134            )
135
136            data[name] = data_model
137
138        # adiciona QMLs brutas
139        data["qmls"] = [ReaderFactory.get_reader(q, SingleHeaderStrategy()).read() for q in qml_files]
140
141        # Adiciona arquivos faltando ou não obrigatórios como vazios
142        for name, (req, _, _) in self.config.file_specs.items():
143            if name not in data:
144                data[name] = DataLoaderModel(
145                    input_folder=str(self.input_dir),
146                    path=Path(name),
147                    df_data=pd.DataFrame(),
148                    read_success=False,
149                )
150
151        return data, errors

18class DataLoaderModel:
19    """
20    Handles the importation and management of data models from a specified file path,
21    providing access to metadata and data for further operations.
22
23    The class is used to manage the details of data model files, encapsulating metadata
24    such as the file name, extension, and path, as well as the associated data stored
25    as a pandas DataFrame. It provides an informative string representation for debugging
26    and logging purposes.
27
28    :ivar input_folder: Path to the folder where the data model file is located.
29    :type input_folder: str
30    :ivar name: The stem (base name without extension) of the file.
31    :type name: str
32    :ivar filename: The full name of the file including the extension.
33    :type filename: str
34    :ivar extension: The file extension.
35    :type extension: str
36    :ivar path: Full path to the file.
37    :type path: Path
38    :ivar df_data: Data extracted from the file as a pandas DataFrame.
39    :type df_data: pd.DataFrame
40    """
41
42    def __init__(
43        self,
44        input_folder: str,
45        path: Path,
46        df_data: pd.DataFrame,
47        read_success: bool = True,
48    ):
49        # SETUP
50        self.input_folder = input_folder
51        self.path = path
52        self.df_data = df_data
53        self.read_success = read_success
54        self.exists_file = self.path.exists() if isinstance(self.path, Path) else False
55
56        # UNPACKING VARIABLES
57        self.name = self.path.stem
58        self.filename = self.path.name
59        self.extension = self.path.suffix
60        self.path = self.path
61        self.header_type = "single" if self.df_data.columns.nlevels == 1 else "double"
62
63    def __str__(self):
64        return (
65            f"DataLoaderModel({self.name}):\n"
66            + f"  input_folder: {self.input_folder}\n"
67            + f"  name: {self.name}\n"
68            + f"  filename: {self.filename}\n"
69            + f"  extension: {self.extension}\n"
70            + f"  path: {self.path}\n"
71            + f"  df_data: \n{self.df_data.head()}\n"
72            + f"  df_data shape: {self.df_data.shape}\n"
73            + f"  df_data columns: {self.df_data.columns}\n"
74            + f"  df_data dtypes: {self.df_data.dtypes}\n"
75            + f"  header_type: {self.header_type}\n"

 10    """
 11    A class to manage localization and translations for the application.
 12    """

 14    def __init__(self, path_locale_dir=None):
 15        """
 16        Initializes the LanguageManager with the given locale directory and default language.
 17
 18        Args:
 19            path_locale_dir (str): Path to the directory containing locale files.
 20        """
 21        self.path_locale_dir = path_locale_dir or Path(__file__).resolve().parents[3] / "static" / "locales"
 22
 23        self.default_language = LanguageEnum.DEFAULT_LANGUAGE.value
 24        self.current_language = None
 25        self.translations = {}
 26
 27        # Pipiline to configure the language
 28        self._congifure_language()
 29        self.supported_languages = LanguageEnum.list_supported_languages()
 30        self._load_translations(self.current_language)
 31
 32    def _congifure_language(self):
 33        store_locale_path = Path(__file__).resolve().parents[4] / ".config" / "store.locale"
 34
 35        with store_locale_path.open("r", encoding="utf-8") as f:
 36            self.current_language = f.read().strip()
 37            if self.current_language not in LanguageEnum.list_supported_languages():
 38                print(f"Invalid current language '{self.current_language}' in store.locale. Falling back to '{LanguageEnum.DEFAULT_LANGUAGE.value}'.")
 39                self.current_language = self.default_language
 40
 41    def _load_translations(self, lang_code):
 42        """Loads translations from the JSON file for a given language."""
 43        filepath = os.path.join(self.path_locale_dir, lang_code, "messages.json")
 44        try:
 45            with open(filepath, "r", encoding="utf-8") as f:
 46                self.translations = json.load(f)
 47            return True
 48        except FileNotFoundError:
 49            print(f"WARNING: Translation file not found for language '{lang_code}': {filepath}")
 50        except json.JSONDecodeError:
 51            print(f"ERROR: Could not decode JSON for language '{lang_code}': {filepath}")
 52        except Exception as e:
 53            print(f"ERROR: Unexpected error loading language '{lang_code}': {e}")
 54        return False
 55
 56    def set_language(self, lang_code):
 57        """
 58        Sets the current language and loads the corresponding translations.
 59
 60        Args:
 61            lang_code (str): Language code to set.
 62
 63        Returns:
 64            bool: True if the language was successfully set, False otherwise.
 65        """
 66        if lang_code in self.supported_languages:
 67            if self._load_translations(lang_code):
 68                self.current_language = lang_code
 69                return True
 70            else:
 71                print(
 72                    self.text(
 73                        "lang_load_error",
 74                        lang=lang_code,
 75                        default_lang=self.default_language,
 76                    )
 77                )
 78                if self._load_translations(self.default_language):
 79                    self.current_language = self.default_language
 80                    return False
 81                return None
 82        else:
 83            print(self.text("invalid_language", default_lang=self.default_language))
 84            if self.current_language != self.default_language:
 85                if self._load_translations(self.default_language):
 86                    self.current_language = self.default_language
 87            return False
 88
 89    def text(self, key, **kwargs):
 90        """
 91        Retrieves the translated string for the given key in the current language.
 92
 93        Args:
 94            key (str): The key of the string to translate.
 95            **kwargs: Keyword arguments for string formatting.
 96
 97        Returns:
 98            str: The translated and formatted string, or a fallback string.
 99        """
100        translation_object = self.translations.get(key)
101        if isinstance(translation_object, dict):
102            text = translation_object.get("message", f"<Message for '{key}' missing in '{self.current_language}'>")
103        else:
104            text = f"<'{key}' missing or invalid structure in '{self.current_language}'>"
105
106        try:
107            return text.format(**kwargs) if kwargs else text
108        except KeyError as e:
109            print(f"Warning: Formatting error for key '{key}'. Missing placeholder: {e}")
110            return text
111        except Exception as format_exc:
112            print(f"Warning: Generic formatting error for key '{key}': {format_exc}")
113            return text
114
115    def get_supported_languages(self):
116        """Returns the list of supported language codes."""
117        return self.supported_languages
118
119    def get_current_language(self):
120        """Returns the currently selected language code."""
121        return self.current_language
122
123    def get_info(self):
124        """
125        Returns the current language and supported languages.
126        """
127        return {
128            "current_language": self.current_language,
129            "supported_languages": self.supported_languages,
130        }
131
132    def __str__(self):
133        """
134        Returns a string representation of the LanguageManager object.
135        """
136        return f"LanguageManager(current_language={self.current_language}, supported_languages={self.supported_languages})"
