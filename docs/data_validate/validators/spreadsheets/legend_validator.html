 95    def validate_relation_indicators_in_legend(self) -> Tuple[List[str], List[str]]:
 96        errors, warnings = [], []
 97
 98        if not self.model_sp_legend.is_sanity_check_passed:
 99            return errors, warnings
100
101        required_columns = {
102            self.sp_name_description: [
103                SpDescription.RequiredColumn.COLUMN_CODE.name,
104                SpDescription.RequiredColumn.COLUMN_LEVEL.name,
105                SpDescription.DynamicColumn.COLUMN_LEGEND.name,
106            ]
107        }
108
109        for column in required_columns[self.sp_name_description]:
110            exists_column, msg_error = self.column_exists(
111                self.model_dataframes[self.sp_name_description],
112                self.sp_name_description,
113                column,
114            )
115            if not exists_column:
116                errors.append(msg_error)
117                break
118
119        if errors:
120            return errors, warnings
121
122        df_legend = self.model_dataframes[self.sp_name_legend].copy()
123        df_description = self.model_dataframes[self.sp_name_description].copy()
124
125        df_description_clean, __ = clean_dataframe_integers(
126            df_description,
127            self.sp_name_description,
128            [SpDescription.RequiredColumn.COLUMN_CODE.name],
129            min_value=1,
130        )
131
132        df_description_clean[SpDescription.DynamicColumn.COLUMN_LEGEND.name] = (
133            pd.to_numeric(
134                df_description_clean[SpDescription.DynamicColumn.COLUMN_LEGEND.name],
135                errors="coerce",
136            )
137        )
138        df_description_clean[SpDescription.DynamicColumn.COLUMN_LEGEND.name] = (
139            df_description_clean[SpDescription.DynamicColumn.COLUMN_LEGEND.name].astype(
140                "Int64"
141            )
142        )
143
144        # Remove all NaN values in column COLUMN_LEGEND
145        df_description_clean = df_description_clean.dropna(
146            subset=[SpDescription.RequiredColumn.COLUMN_CODE.name]
147        )
148        df_legend = df_legend.dropna(subset=[SpLegend.RequiredColumn.COLUMN_CODE.name])
149
150        legends_id_in_description = (
151            df_description_clean[
152                (
153                    df_description_clean[SpDescription.RequiredColumn.COLUMN_LEVEL.name]
154                    != "1"
155                )
156                & (
157                    df_description_clean[
158                        SpDescription.DynamicColumn.COLUMN_LEGEND.name
159                    ].notna()
160                )
161                & (
162                    pd.to_numeric(
163                        df_description_clean[
164                            SpDescription.DynamicColumn.COLUMN_LEGEND.name
165                        ],
166                        errors="coerce",
167                    ).notna()
168                )
169            ][SpDescription.DynamicColumn.COLUMN_LEGEND.name]
170            .astype(str)
171            .tolist()
172        )  # DROP NA VALUES
173
174        legends_id_in_legend = (
175            df_legend[SpLegend.RequiredColumn.COLUMN_CODE.name]
176            .astype(str)
177            .unique()
178            .tolist()
179        )
180
181        set_one = set(legends_id_in_description)
182        set_two = set(legends_id_in_legend)
183
184        missing_in_b, missing_in_a = find_differences_in_two_set(
185            first_set=set_one,
186            second_set=set_two,
187        )
188
189        missing_in_b = {int(x) for x in missing_in_b if str(x).isdigit()}
190        missing_in_a = {int(x) for x in missing_in_a if str(x).isdigit()}
191
192        if missing_in_b:
193            errors.append(
194                f"{self.sp_name_description}: Códigos de legenda ausentes em {self.sp_name_legend}: {sorted(list(missing_in_b))}."
195            )
196
197        if missing_in_a:
198            warnings.append(
199                f"{self.sp_name_legend}: Códigos de legenda não referenciados em {self.sp_name_description}: {sorted(list(missing_in_a))}."
200            )
201
202        # 1. All codes that are level 1 - Cannot have legends: if they do, error
203
204        codes_indicators_level_one = (
205            df_description_clean[
206                df_description_clean[SpDescription.RequiredColumn.COLUMN_LEVEL.name]
207                == "1"
208            ][SpDescription.RequiredColumn.COLUMN_CODE.name]
209            .astype(str)
210            .tolist()
211        )
212
213        legends_id_in_description_level_one = (
214            df_description_clean[
215                (
216                    df_description_clean[SpDescription.RequiredColumn.COLUMN_LEVEL.name]
217                    == "1"
218                )
219                & (
220                    df_description_clean[
221                        SpDescription.DynamicColumn.COLUMN_LEGEND.name
222                    ].notna()
223                )
224            ][SpDescription.DynamicColumn.COLUMN_LEGEND.name]
225            .unique()
226            .astype(str)
227            .tolist()
228        )
229
230        if legends_id_in_description_level_one:
231            errors.append(
232                f"{self.sp_name_description}: Indicadores de nível 1 não podem ter referência de legenda. Códigos com referência em {self.sp_name_legend}: {sorted(list(codes_indicators_level_one))}."
233            )
234
235        # 3. All codes that are not level 1 and not level 2 - Must have a legend reference: if not, error
236        codes_indicators_other_levels = (
237            df_description_clean[
238                (
239                    df_description_clean[SpDescription.RequiredColumn.COLUMN_LEVEL.name]
240                    != "1"
241                )
242                & (
243                    df_description_clean[SpDescription.RequiredColumn.COLUMN_LEVEL.name]
244                    != "2"
245                )
246            ][SpDescription.RequiredColumn.COLUMN_CODE.name]
247            .astype(str)
248            .tolist()
249        )
250
251        codes_with_legend_other_levels = (
252            df_description_clean[
253                (
254                    df_description_clean[SpDescription.RequiredColumn.COLUMN_LEVEL.name]
255                    != "1"
256                )
257                & (
258                    df_description_clean[SpDescription.RequiredColumn.COLUMN_LEVEL.name]
259                    != "2"
260                )
261                & (
262                    df_description_clean[
263                        SpDescription.DynamicColumn.COLUMN_LEGEND.name
264                    ].notna()
265                )
266            ][SpDescription.RequiredColumn.COLUMN_CODE.name]
267            .astype(str)
268            .tolist()
269        )
270
271        set_codes_other_levels = set(codes_indicators_other_levels)
272        set_codes_with_legend = set(codes_with_legend_other_levels)
273        missing_legends_other_levels = set_codes_other_levels - set_codes_with_legend
274
275        if missing_legends_other_levels:
276            errors.append(
277                f"{self.sp_name_description}: Indicadores de níveis diferentes de 1 e 2 devem ter referência de legenda. Indicadores sem referência em {self.sp_name_legend}: {sorted(list(missing_legends_other_levels))}."
278            )
279
280        return errors, warnings

282    def validate_range_multiple_legend(self) -> Tuple[List[str], List[str]]:
283        errors, warnings = [], []
284
285        if self.model_sp_value.data_loader_model.df_data.empty:
286            return errors, warnings
287
288        min_lower_legend_default = (
289            self.model_sp_legend.CONSTANTS.MIN_LOWER_LEGEND_DEFAULT
290        )
291        max_upper_legend_default = (
292            self.model_sp_legend.CONSTANTS.MAX_UPPER_LEGEND_DEFAULT
293        )
294        required_columns = {
295            self.sp_name_description: [
296                SpDescription.RequiredColumn.COLUMN_CODE.name,
297                SpDescription.RequiredColumn.COLUMN_LEVEL.name,
298            ]
299        }
300
301        if self.model_sp_legend.legend_read_success:
302            required_columns[self.sp_name_description].append(
303                SpDescription.DynamicColumn.COLUMN_LEGEND.name
304            )
305
306        for column in required_columns[self.sp_name_description]:
307            exists_column, msg_error = self.column_exists(
308                self.model_dataframes[self.sp_name_description],
309                self.sp_name_description,
310                column,
311            )
312            if not exists_column:
313                errors.append(msg_error)
314                break
315
316        if errors:
317            return errors, warnings
318
319        df_values = self.model_dataframes[self.sp_name_value].copy()
320        df_legend = self.model_dataframes[self.sp_name_legend].copy()
321        df_description = self.model_dataframes[self.sp_name_description].copy()
322
323        if SpValue.RequiredColumn.COLUMN_ID.name in df_values.columns:
324            df_values.drop(
325                columns=[SpValue.RequiredColumn.COLUMN_ID.name], inplace=True
326            )
327
328        df_description_clean, __ = clean_dataframe_integers(
329            df_description,
330            self.sp_name_description,
331            [SpDescription.RequiredColumn.COLUMN_CODE.name],
332            min_value=1,
333        )
334
335        if (
336            SpDescription.DynamicColumn.COLUMN_LEGEND.name
337            not in df_description_clean.columns
338        ):
339            df_description_clean[SpDescription.DynamicColumn.COLUMN_LEGEND.name] = (
340                pd.Series(dtype="Int64")
341            )
342        else:
343            df_description_clean[SpDescription.DynamicColumn.COLUMN_LEGEND.name] = (
344                pd.to_numeric(
345                    df_description_clean[
346                        SpDescription.DynamicColumn.COLUMN_LEGEND.name
347                    ],
348                    errors="coerce",
349                )
350            )
351            df_description_clean[SpDescription.DynamicColumn.COLUMN_LEGEND.name] = (
352                df_description_clean[
353                    SpDescription.DynamicColumn.COLUMN_LEGEND.name
354                ].astype("Int64")
355            )
356
357        codes_indicators_level_one = (
358            df_description_clean[
359                df_description_clean[SpDescription.RequiredColumn.COLUMN_LEVEL.name]
360                == "1"
361            ][SpDescription.RequiredColumn.COLUMN_CODE.name]
362            .astype(str)
363            .tolist()
364        )
365        valid_columns_from_values, __ = categorize_strings_by_id_pattern_from_list(
366            items_to_categorize=df_values.columns.tolist(),
367            allowed_scenario_suffixes=self.scenarios_list,
368        )
369
370        groups_legends = pd.DataFrame(
371            {str(SpLegend.RequiredColumn.COLUMN_CODE.name): []}
372        ).groupby(str(SpLegend.RequiredColumn.COLUMN_CODE.name))
373        if self.model_sp_legend.is_sanity_check_passed:
374            groups_legends = df_legend.groupby(
375                str(SpLegend.RequiredColumn.COLUMN_CODE.name)
376            )
377
378        mapping_legends = {}
379        for data_column_sp_value in valid_columns_from_values:
380            aux_indicator_id = data_column_sp_value.split("-")[0]
381            aux_data_mapping_legend = ModelMappingLegend(
382                column_sp_value=data_column_sp_value,
383                default_min_value=min_lower_legend_default,
384                default_max_value=max_upper_legend_default,
385            )
386
387            if aux_indicator_id in codes_indicators_level_one:
388                continue
389
390            row_description = df_description_clean[
391                df_description_clean[
392                    SpDescription.RequiredColumn.COLUMN_CODE.name
393                ].astype(str)
394                == aux_indicator_id
395            ]
396            if not row_description.empty:
397                aux_data_mapping_legend.indicator_id = aux_indicator_id
398
399                key_legend = row_description.iloc[0][
400                    SpDescription.DynamicColumn.COLUMN_LEGEND.name
401                ]
402                group_legend = (
403                    groups_legends.get_group(str(key_legend))
404                    if str(key_legend) in groups_legends.groups
405                    else None
406                )
407
408                if group_legend is not None:
409                    aux_data_mapping_legend.legend_id = key_legend
410
411                    group_legend = group_legend[
412                        group_legend[SpLegend.RequiredColumn.COLUMN_LABEL.name]
413                        != self._data_models_context.config.VALUE_DATA_UNAVAILABLE
414                    ]
415                    if not group_legend.empty:
416                        group_legend[SpLegend.RequiredColumn.COLUMN_MINIMUM.name] = (
417                            pd.to_numeric(
418                                group_legend[
419                                    SpLegend.RequiredColumn.COLUMN_MINIMUM.name
420                                ],
421                                errors="coerce",
422                            )
423                        )
424                        group_legend[SpLegend.RequiredColumn.COLUMN_MAXIMUM.name] = (
425                            pd.to_numeric(
426                                group_legend[
427                                    SpLegend.RequiredColumn.COLUMN_MAXIMUM.name
428                                ],
429                                errors="coerce",
430                            )
431                        )
432
433                        aux_min_value = group_legend[
434                            SpLegend.RequiredColumn.COLUMN_MINIMUM.name
435                        ].min()
436                        aux_max_value = group_legend[
437                            SpLegend.RequiredColumn.COLUMN_MAXIMUM.name
438                        ].max()
439
440                        if not pd.isna(aux_min_value) and not pd.isna(aux_max_value):
441                            aux_data_mapping_legend.min_value = aux_min_value
442                            aux_data_mapping_legend.max_value = aux_max_value
443
444            mapping_legends[data_column_sp_value] = aux_data_mapping_legend
445
446        df_values_numeric = df_values.copy()
447        for col in valid_columns_from_values:
448            df_values_numeric[col] = pd.to_numeric(
449                df_values[col].astype(str).str.replace(",", "."), errors="coerce"
450            )
451
452        for data_column_sp_value in valid_columns_from_values:
453            code_column = data_column_sp_value.split("-")[0]
454
455            if code_column in codes_indicators_level_one:
456                continue
457
458            min_value = mapping_legends[data_column_sp_value].min_value
459            max_value = mapping_legends[data_column_sp_value].max_value
460
461            for index, value_numeric in df_values_numeric[data_column_sp_value].items():
462                value_original = df_values[data_column_sp_value][index]
463                if (
464                    value_original == self._data_models_context.config.VALUE_DI
465                    or pd.isna(value_numeric)
466                ):
467                    continue
468
469                if value_numeric < min_value or value_numeric > max_value:
470                    errors.append(
471                        f"{self.sp_name_value}, linha {index + 2}: O valor {value_original} está fora do intervalo da legenda ({min_value} a {max_value}) para a coluna '{data_column_sp_value}'."
472                    )
473
474        return errors, warnings

476    def run(self) -> Tuple[List[str], List[str]]:
477        """Runs all content validations for SpLegend."""
478        validations = []
479
480        if self.model_sp_legend.is_sanity_check_passed:
481            validations.append(
482                (self.validate_relation_indicators_in_legend, NamesEnum.LEG_REL.value)
483            )
484        validations.append(
485            (self.validate_range_multiple_legend, NamesEnum.LEG_RANGE.value)
486        )
487
488        if self.model_sp_description.data_loader_model.df_data.empty:
489            self.set_not_executed(validations)
490            return self._errors, self._warnings
491
492        # BUILD REPORTS
493        self.build_reports(validations)
494
495        return self._errors, self._warnings
